السبب: عدم وجود رسم فوري عند إنشاء مستويات فيبوناتشي، والاعتماد فقط على دورة الرسم التلقائية.

🚀رسم فوري للنقرة الأولى

إضافة رسم فوري لنقطة البداية عند النقرة الأولى:

// ✅ رسم فوري لنقطة البداية const canvas = canvasRef.current; if (canvas) { const ctx = canvas.getContext('2d'); ctx.save(); // رسم نقطة البداية المتوهجة ctx.fillStyle = '#00FFFF'; ctx.shadowColor = '#00FFFF'; ctx.shadowBlur = 10; ctx.beginPath(); ctx.arc(x, y, 8, 0, Math.PI * 2); ctx.fill(); // رسم دائرة خارجية ctx.strokeStyle = '#FFFFFF'; ctx.lineWidth = 2; ctx.shadowBlur = 0; ctx.beginPath(); ctx.arc(x, y, 12, 0, Math.PI * 2); ctx.stroke(); ctx.restore(); }

📏معاينة مباشرة أثناء الحركة

إضافة خط معاينة يتبع الماوس بين النقرتين:

// ✅ رسم خط المعاينة المتحرك if (interactiveMode === 'fibonacci') { const ctx = canvas.getContext('2d'); // رسم خط المعاينة ctx.save(); ctx.strokeStyle = '#FFD700'; ctx.lineWidth = 2; ctx.setLineDash([3, 3]); ctx.globalAlpha = 0.8; ctx.beginPath(); ctx.moveTo(drawingStart.x, drawingStart.y); ctx.lineTo(x, y); ctx.stroke(); // رسم نقطة النهاية المؤقتة ctx.fillStyle = '#FFD700'; ctx.globalAlpha = 0.6; ctx.beginPath(); ctx.arc(x, y, 6, 0, Math.PI * 2); ctx.fill(); ctx.restore(); }

🎯رسم فوري للمستويات

رسم مستويات فيبوناتشي فوراً عند النقرة الثانية:

// ✅ رسم فوري لمستويات فيبوناتشي fibLevels.forEach(level => { // رسم خط المستوى ctx.beginPath(); ctx.moveTo(level.startX, level.y); ctx.lineTo(level.endX, level.y); ctx.stroke(); // رسم النص والنسبة ctx.fillStyle = INTERACTIVE_TOOLS.fibonacci.color; ctx.font = '12px Arial'; ctx.fillText(`${level.ratio.toFixed(1)}%`, level.endX + 5, level.y - 5); });

🔄useEffect للرسم التلقائي

إضافة useEffect لرسم المستويات عند تحديث البيانات:

// ✅ useEffect للرسم الفوري useEffect(() => { const canvas = canvasRef.current; if (!canvas || fibonacciLevels.length === 0) return; const redrawFibonacci = () => { const ctx = canvas.getContext('2d'); fibonacciLevels.forEach(fibGroup => { // رسم جميع مستويات المجموعة fibGroup.levels.forEach(level => { // رسم خط المستوى + النص + النقاط }); }); }; requestAnimationFrame(redrawFibonacci); }, [fibonacciLevels]);
